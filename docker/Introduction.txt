#==========================BASE DE DATOS==========================
#Para ingresar a la consola de la base de datos
    docker compose exec mysql sh
    mysql -u root -p latinpaychallenge
    USE latinpaychallenge;
    SHOW TABLES;

#Con esto generamos el backup de la base de datos desde el mismo lugar donde lo ejecutas
    #Tiene que existir la carpeta backups/ para que funcione
    #docker exec -t laravel_mysql /usr/bin/mysqldump -u root --password=root latinpaychallenge > backups/backup.sql  (Funciona pero agrega advertencias de contraseña)
    docker exec -i -e MYSQL_PWD=root laravel_mysql /usr/bin/mysqldump -u root latinpaychallenge > backups/backup.sql (ELIGE ESTO POR QUE EN EL PRIMERO TE AGREGA ADVERTENCIAS DE CONTRASEÑA)

    #Con fecha
    docker exec -i -e MYSQL_PWD=root laravel_mysql mysqldump -u root latinpaychallenge > backups/backup-$(date +%Y-%m-%d_%H-%M-%S).sql (Para usar este tienes que estar dentro del terminal de linux, ejecutarlo desde powershell da error porque no reconoce el comando date)
    docker exec -t laravel_mysql `mysqldump -u root --password=root latinpaychallenge` > "backups/backup-$(Get-Date -Format 'yyyy-MM-dd_HH-mm-ss').sql" (Para usar este tienes que estar dentro de powershell)
